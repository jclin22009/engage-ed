<script>
let button = document.getElementById("chosenSpell");
let correctAnswer = document.getElementById("spelling2");
button.addEventListener("click", evaluateCorrectness);
function evaluateCorrectness() {
    let output = document.getElementById("output");
    let spells = document.getElementsByName("spelling");
    for (let x of spells) {
        if (x.checked) {
            if (x == correctAnswer) {
                output.innerText = "Correct" 
            } else {
                output.innerText = "Incorrect"
            }
        }
    }
}
</script>
